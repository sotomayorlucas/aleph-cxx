- [ ] **Step 2 — branchless ONB** (in `detail`, before `ambient_occlusion`):
```cpp
// Duff et al. 2017 branchless orthonormal basis from a unit normal n.
inline void onb_from_normal(aleph::math::Vec3 n,
                            aleph::math::Vec3& T, aleph::math::Vec3& B) noexcept {
    const aleph::math::f32 s = std::copysign(1.0f, n.z);
    const aleph::math::f32 a = -1.0f / (s + n.z);
    const aleph::math::f32 b = n.x * n.y * a;
    T = aleph::math::Vec3{1.0f + s * n.x * n.x * a, s * b, -s * n.x};
    B = aleph::math::Vec3{b, s + n.y * n.y * a, -n.y};
}
```

- [ ] **Step 3 — `ambient_occlusion`** (in `detail`, after the `seg_hits_*` + before/near `shade_face`):
```cpp
[[nodiscard]] inline aleph::math::f32
ambient_occlusion(aleph::math::Vec3 point, aleph::math::Vec3 N,
                  const std::vector<LoweredEntity>& occluders,
                  aleph::types::NodeId self) noexcept {
    using aleph::math::Vec3;
    // Reorient to the world-up hemisphere: the quad/tri normal sign is arbitrary
    // (the floor's cross(u,v) points DOWN); AO is sign-sensitive, so sample the
    // upward hemisphere. (Lambert keeps the raw N — this flip is local to AO.)
    const Vec3 N_ao = (aleph::math::dot(N, Vec3{0.0f, 1.0f, 0.0f}) < 0.0f) ? N * -1.0f : N;
    Vec3 T, B; onb_from_normal(N_ao, T, B);
    const Vec3 p0 = point + N_ao * kAoBias;
    int occluded = 0;
    for (const Vec3& dt : kAoDirs) {
        const Vec3 d = T * dt.x + B * dt.y + N_ao * dt.z;   // world dir (unit)
        bool hit = false;
        for (const LoweredEntity& o : occluders) {
            if (o.source == self) continue;
            if (std::visit([&](const auto& gg) {
                    using G = std::decay_t<decltype(gg)>;
                    if constexpr (std::is_same_v<G, aleph::types::SphereLocal>) return seg_hits_sphere(p0, d, kAoDist, gg);
                    else if constexpr (std::is_same_v<G, aleph::types::QuadLocal>) return seg_hits_quad(p0, d, kAoDist, gg);
                    else return seg_hits_tri(p0, d, kAoDist, gg);
                }, o.world_geometry)) { hit = true; break; }
        }
        if (hit) ++occluded;
    }
    return std::max(kAoFloor,
        1.0f - static_cast<aleph::math::f32>(occluded) / static_cast<aleph::math::f32>(kAoRays));
}
```
(`<cmath>` for `std::copysign` is already in the global fragment.)
